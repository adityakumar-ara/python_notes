def check_even_odd(num):
    if num % 2 == 0:
        return "Even"
    else:
        return "Odd"

number = int(input("Enter a number: "))

result = check_even_odd(number)
print(f"The number {number} is {result}.")



Example Output
Enter a number: 9
The number 9 is Odd.
